class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age

    def greet(self):
        print(f"Hello, my name is {self.name} and I am {self.age} years old.")

def main():
    person1 = Person("Alice", 25)
    person1.greet()

if __name__ == "__main__":
    main()
